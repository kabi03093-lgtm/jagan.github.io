const lines = [
"To automatically move text from left to right across the screen,",
"you can use CSS animations with @keyframes.",

];

let lineIndex = 0;
let charIndex = 0;
let speed = 50; // typing speed (ms)

function typeText() {
    if (lineIndex < lines.length) {
        if (charIndex < lines[lineIndex].length) {
            document.getElementById("text").innerHTML += lines[lineIndex][charIndex];
            charIndex++;
            setTimeout(typeText, speed);
        } else {
            document.getElementById("text").innerHTML += "<br>";
            lineIndex++;
            charIndex = 0;
            setTimeout(typeText, 500); // delay between lines
        }
    }
}
